	INPUT
	STORE	X	\{ENTRA E ARMAZENA X
	LOAD	X	\CARREGA X
	ADD	X	 \ADICIONA ELE MESMO
	STORE	NOVO	 \NOVA VARIAVEL SALVA DO RESULTADO
	OUTPUT		  \SAIDA DO VALOR}

LOOP,	LOAD	X	  \{CARREGA O VALOR QUE ENTROU	\ LOOP QUE SOMA ATÉ ZERAR
	ADD	NOVO	  \ADICIONA O RESULTADO DA PRIMEIRA FUNÇÃO	
	STORE	NOVO	   \ARMAZENA ESSA NOVA VARIÁVEL	
	OUTPUT		   \SAIDA DESSE NOVO 2º VALOR ENCONTRADO}	

	LOAD	CONT	    \{CARREGA UMA NOVA VARIÁVEL DE CONTADOR
	SUBT	TIRA	    \SUBTRAI A VARIAVEL POR 1
	STORE	CONT	     \ARMAZENA ESSA NOVA VARIÁVEL}
	SKIPCOND 400	     \SE FOR IGUAL A ZERO ELE PULA O JUMP E ENCERRA ENQUANTO NÃO FOR VOLTA PARA O LOOP
	JUMP	LOOP	     \O LOOP VOLTA PARA O LOAD X DE NOVO ATÉ QUE ZERE O SCKIPCOND
	HALT		     \QUANDO DER ZERO ENCERRA
CONT,	DEC	8	     \VARIÁVEL DO CONTADOR
TIRA,	DEC	1	     \VARIÁVEL QUE ESTÁ TIRANDO DO CONTADOR
NUM,	DEC	0	     \VARIAVEL QUE USUÁRIO ENTRA, TEM QUE SER ZERO PARA NÃO TER NADA NA MEMÓRIA
NOVO,	DEC	0	      \VARIÁVEL DO NOVO RESULTADO DO 1º BLOCO, TEM QUE SER ZERO P/ NÃO TER NADA NA MEMÓRIA
